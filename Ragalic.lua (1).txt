<script>
    // ===== ИГРОВОЕ СОСТОЯНИЕ =====
    const state = {
        level: 1,
        exp: 0,
        expToNext: 100,
        totalLifts: 0,
        power: 1,
        mode: 'normal', // normal | exploit | auto | mass
        exploitActive: false,
        massActive: false,
        autoActive: false,
        autoInterval: null,
        massHitCount: 0,
    };

    // ===== DOM =====
    const levelDisplay = document.getElementById('levelDisplay');
    const expDisplay = document.getElementById('expDisplay');
    const progressFill = document.getElementById('progressFill');
    const liftBtn = document.getElementById('liftBtn');
    const powerDisplay = document.getElementById('powerDisplay');
    const liftsDisplay = document.getElementById('liftsDisplay');
    const exploitDisplay = document.getElementById('exploitDisplay');
    const massDisplay = document.getElementById('massDisplay');
    const modeLabel = document.getElementById('modeLabel');
    const statusMode = document.getElementById('statusMode');
    const autoStatus = document.getElementById('autoStatus');
    const massStatus = document.getElementById('massStatus');
    const autoDot = document.getElementById('autoDot');
    const massDot = document.getElementById('massDot');
    const hitIndicator = document.getElementById('hitIndicator');
    const massNotify = document.getElementById('massNotify');

    const menuBtns = document.querySelectorAll('.menu-btn');

    // ===== ОБНОВЛЕНИЕ UI =====
    function updateUI() {
        levelDisplay.textContent = state.level;
        expDisplay.textContent = `${Math.floor(state.exp)} / ${state.expToNext}`;
        const percent = Math.min((state.exp / state.expToNext) * 100, 100);
        progressFill.style.width = `${percent}%`;
        powerDisplay.textContent = state.power;
        liftsDisplay.textContent = state.totalLifts;
        exploitDisplay.textContent = state.exploitActive ? 'ON' : 'OFF';
        exploitDisplay.style.color = state.exploitActive ? '#ff6b6b' : '#6f8bb0';
        massDisplay.textContent = state.massActive ? 'ON' : 'OFF';
        massDisplay.style.color = state.massActive ? '#bb86fc' : '#6f8bb0';

        // Обновляем статус
        statusMode.textContent = state.mode.toUpperCase();
        modeLabel.textContent = state.mode === 'mass' ? '💀 MASS HIT' : 
                               state.mode === 'exploit' ? '⚡ ЭКСПЛОИТ' :
                               state.mode === 'auto' ? '🔄 АВТО' : 'NORMAL';
        
        if (state.mode === 'mass') {
            modeLabel.classList.add('active-mass');
        } else {
            modeLabel.classList.remove('active-mass');
        }

        autoStatus.textContent = state.autoActive ? 'Авто: вкл' : 'Авто: выкл';
        autoDot.className = `dot ${state.autoActive ? 'green' : ''}`;
        
        massStatus.textContent = state.massActive ? 'Mass: вкл' : 'Mass: выкл';
        massDot.className = `dot ${state.massActive ? 'red' : ''}`;

        // Индикатор на кнопке
        if (state.massActive) {
            hitIndicator.textContent = '💀 MASS';
            hitIndicator.style.color = '#ff4444';
        } else if (state.exploitActive) {
            hitIndicator.textContent = '⚡ x2';
            hitIndicator.style.color = '#ff6b6b';
        } else {
            hitIndicator.textContent = '🎯';
            hitIndicator.style.color = '#ff4444';
        }
    }

    // ===== ОСНОВНАЯ МЕХАНИКА ПОДЪЁМА =====
    function performLift() {
        let expGain = state.power;

        // Эксплоит: удвоение
        if (state.exploitActive) {
            expGain *= 2;
        }

        // Mass Hit: удар по всем на сервере (дополнительный бонус)
        if (state.massActive) {
            // Эффект масс-хита — добавляем бонус за каждого "игрока" на сервере
            // Имитация: +50% за каждого из 10 "игроков"
            const serverPlayers = 10;
            const massBonus = state.power * 1.5 * serverPlayers;
            expGain += massBonus;
            state.massHitCount++;
            
            // Показываем уведомление о масс-хите (не спамим)
            if (state.massHitCount % 3 === 0) {
                showMassNotify();
            }
        }

        state.exp += expGain;
        state.totalLifts++;

        // Проверка на повышение уровня
        while (state.exp >= state.expToNext) {
            state.exp -= state.expToNext;
            state.level++;
            state.expToNext = Math.floor(state.expToNext * 1.5) + 50;
            state.power = Math.floor(state.power * 1.2) + 1;
            
            // При повышении уровня в масс-хите — дополнительный буст
            if (state.massActive) {
                state.power += 5;
            }
        }

        // Анимация кнопки
        liftBtn.classList.remove('lift-flash');
        void liftBtn.offsetWidth;
        liftBtn.classList.add('lift-flash');

        updateUI();
    }

    // ===== УВЕДОМЛЕНИЕ О MASS HIT =====
    let notifyTimeout = null;

    function showMassNotify() {
        massNotify.style.display = 'block';
        clearTimeout(notifyTimeout);
        notifyTimeout = setTimeout(() => {
            massNotify.style.display = 'none';
        }, 2000);
    }

    // ===== АВТО-ЛИФТ =====
    function toggleAuto() {
        if (state.autoActive) {
            clearInterval(state.autoInterval);
            state.autoActive = false;
            state.autoInterval = null;
        } else {
            state.autoActive = true;
            state.autoInterval = setInterval(() => {
                performLift();
            }, 300);
        }
        updateUI();
    }

    // ===== ПЕРЕКЛЮЧЕНИЕ РЕЖИМОВ =====
    function setMode(mode) {
        // Выключаем авто если включен
        if (state.autoActive) {
            clearInterval(state.autoInterval);
            state.autoActive = false;
            state.autoInterval = null;
        }

        state.mode = mode;

        // Сбрасываем активные режимы
        state.exploitActive = false;
        state.massActive = false;

        switch (mode) {
            case 'normal':
                // Всё выключено
                break;
            case 'exploit':
                state.exploitActive = true;
                break;
            case 'auto':
                state.autoActive = true;
                state.autoInterval = setInterval(() => {
                    performLift();
                }, 300);
                break;
            case 'mass':
                state.massActive = true;
                state.exploitActive = true; // Комбо: эксплоит + масс
                // Автоматический масс-хит каждые 2 секунды
                if (!state.autoInterval) {
                    state.autoActive = true;
                    state.autoInterval = setInterval(() => {
                        performLift();
                    }, 250);
                }
                break;
        }

        // Обновляем активные кнопки
        menuBtns.forEach(btn => {
            btn.classList.remove('active');
            if (btn.dataset.mode === mode) {
                btn.classList.add('active');
            }
        });

        updateUI();
    }

    // ===== СБРОС =====
    function resetGame() {
        if (state.autoActive) {
            clearInterval(state.autoInterval);
            state.autoActive = false;
            state.autoInterval = null;
        }
        
        state.level = 1;
        state.exp = 0;
        state.expToNext = 100;
        state.totalLifts = 0;
        state.power = 1;
        state.massHitCount = 0;
        state.exploitActive = false;
        state.massActive = false;
        state.mode = 'normal';
        
        menuBtns.forEach(btn => {
            btn.classList.remove('active');
            if (btn.dataset.mode === 'normal') {
                btn.classList.add('active');
            }
        });
        
        updateUI();
    }

    // ===== СОБЫТИЯ =====
    liftBtn.addEventListener('click', (e) => {
        // Масс-хит активируется при клике в любом месте кнопки
        performLift();
    });

    // Клик по кнопкам меню
    menuBtns.forEach(btn => {
        btn.addEventListener('click', () => {
            const mode = btn.dataset.mode;
            setMode(mode);
        });
    });

    document.getElementById('resetBtn').addEventListener('click', resetGame);

    // ===== КЛИК В ЛЮБОМ МЕСТЕ СТРАНИЦЫ (для масс-хита) =====
    document.addEventListener('click', (e) => {
        // Если режим масс-хит активен — каждый клик = удар по всем
        if (state.massActive) {
            // Дополнительный масс-хит при клике в любом месте
            // Но не спамим слишком часто
            performLift();
        }
